# requirements.R
# Install required R packages for the project

packages <- c(
  "forecast",   # SARIMA modeling and forecasting
  "fpp3",       # time series analysis framework
  "tseries",    # stationarity tests, ADF, Ljung-Box
  "ggplot2",    # visualization
  "dplyr",      # data manipulation
  "tsibble",    # tidy time series data
  "lubridate",  # date handling
  "knitr",      # reproducible reports
  "rmarkdown"   # R Markdown support
)

install_if_missing <- function(p) {
  if (!requireNamespace(p, quietly = TRUE)) {
    install.packages(p, repos = "https://cloud.r-project.org")
  }
}

invisible(lapply(packages, install_if_missing))
